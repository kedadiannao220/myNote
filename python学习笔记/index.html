其他类库

xlrd : python读取excel

import xlrd

if __name__ == '__main__':
    data = xlrd.open_workbook("/tmp/test.xlsx")
    sheet = data.sheets()[0]
    nrows = sheet.nrows
    ncols = sheet.ncols
    for rowNum in range(0, nrows):
        row = sheet.row_values(rowNum)
        for colNum in range(0, ncols):
            cell = row[colNum]
            print(cell)
