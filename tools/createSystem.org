*** kvm virt执行脚本
virt-install \
--name=Centos-7 \  #虚拟机名称
--vcpus=4 \        #cpu数量
--ram=4096 \       #内存大小
--hvm \       
--accelerate \
--network=bridge:br0,model=virtio \    #网络是否为桥接
--disk path=/dev/VolGroup0/centos7,format=raw,bus=virtio \  #硬盘的位置，格式化信息
--cdrom=/home/www/iso/CentOS-7-x86_64-DVD-1511.iso \   #镜像地址
--vnc
